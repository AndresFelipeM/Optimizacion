# MEZCLA DE FERTILIZANTES 

from pulp import *

# Datos
disponible_compuesto_A = 120  
disponible_compuesto_B = 80   
disponible_compuesto_C = 150 

# Definir el problema
prob = LpProblem("Mezcla_Fertilizantes", LpMaximize)

# Variables de decision
  # f1 = toneladas de Fertilizante 1 a producir
  # f2 = toneladas de Fertilizante 2 a producir
f1 = LpVariable("F1", lowBound=0)
f2 = LpVariable("F2", lowBound=0)

# Funcion objetivo: Maximizar ganancia
  # f(x1, x2) = 180*x1 + 220*x2
prob += 180*f1 + 220*f2, "Ganancia_Total"

# Restricciones de disponibilidad de compuestos
prob += 3*f1 + 4*f2 <= disponible_compuesto_A, "Compuesto_A"
prob += 2*f1 + 1*f2 <= disponible_compuesto_B, "Compuesto_B"
prob += 5*f1 + 3*f2 <= disponible_compuesto_C, "Compuesto_C"

# Resolver
prob.solve(PULP_CBC_CMD(msg=0))

# Resultados
f1_opt = f1.varValue
f2_opt = f2.varValue
ganancia_max = value(prob.objective)

print(" = " * 45)
print(" Mezcla de Fertilizantes - Maxima Ganancia")
print(" = " * 45)

print(f"Estado: {LpStatus[prob.status]}")
print()
print(f"Produccion F1 : f1 = {f1_opt:.2f} toneladas")
print(f"Produccion F2 : f2 = {f2_opt:.2f} toneladas")
print(f"Ganancia maxima      : $ {ganancia_max:.2f}")

print(" = " * 45)

# Verificacion de restricciones
uso_A = 3*f1_opt + 4*f2_opt
uso_B = 2*f1_opt + 1*f2_opt
uso_C = 5*f1_opt + 3*f2_opt

print("Verificacion de uso de compuestos:")
print(f"  Compuesto A: {uso_A:.2f} kg usados de {disponible_compuesto_A} kg disponibles (restante: {disponible_compuesto_A - uso_A:.2f} kg)")
print(f"  Compuesto B: {uso_B:.2f} kg usados de {disponible_compuesto_B} kg disponibles (restante: {disponible_compuesto_B - uso_B:.2f} kg)")
print(f"  Compuesto C: {uso_C:.2f} kg usados de {disponible_compuesto_C} kg disponibles (restante: {disponible_compuesto_C - uso_C:.2f} kg)")
print(" = " * 45)
